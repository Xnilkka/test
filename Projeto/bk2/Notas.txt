package Trabalho;

public class Notas {
	//Atributos
	private float nota1 = -1;
	private float nota2 = -1;
	
	//Metodos
	
	public Notas(float nota1, float nota2) {
		this.nota1 = nota1;
		this.nota2 = nota2;
	}
	
	

	public float getNota1() {
		return nota1;
	}

	public void setNota1(float nota1) {
		this.nota1 = nota1;
	}

	public float getNota2() {
		return nota2;
	}

	public void setNota2(float nota2) {
		this.nota2 = nota2;
	}
}
